1. var = before ES6 / globle scope

eg:

var a; // Declaration
var a = "abc"// Declaration and initialization
var a = "abc" // Re-Declaration is possible
name = "Khyati Patel" // Re-Assignament is possible
console.log(a) .... {Ans - abc}
console.log(name) .... {Ans - Khyati Patel}
console.log(a,name) .... {Ans -abc, Khyati Patel}

eg:
