DataType

two type datatype
  • Premative
  • Non-premative

1. Premative

  1. Boolean

    2. eg :

    let a = false
    console.log(a, typeof a);
    ans : false 'boolean'

    let aa = true
    console.log(aa, typeof aa);
    ans : true 'boolean'

  2. Bigint

    3. eg :

    let b = 5n
    console.log(b, typeof b);
    ans : 5n 'bigint' ..(only n)

  3. Number

    4. eg :

    let c = 56
    console.log(c, typeof c);
    ans : 56 'number'

  4. Null

    5. eg :

    let d = null
    console.log(d, typeof d);
    ans : null 'object' (null)

  5. Symbol

    6. eg :

    let e = Symbol("name")
    console.log(e, typeof e);
    ans : Symbol(name) 'symbol'

  6. String

    7. eg :

    let f = "Hello"
    console.log(e, typeof e);
    ans : Hello 'string'

  7. Undefiend

    8. eg :

    let g;
    console.log(g, typeof g);
    ans : undefiend 'undefiend'

2. Non-premative

  1. Object

    2. eg :

    let state = {
    Gujarat : 1,
    Rajasthan : 2
    };
    console.log(state, typeof state);
    ans : {Gujarat : 1,Rajasthan : 2} object.

    eg :

    let state = {
    Gujarat
    Rajasthan
    };
    console.log(state, typeof state);
    ans : error

    eg :

    let state = {
    1 : Gujarat
    2 : Rajasthan
    };
    console.log(state, typeof state);
    ans : error

    eg :

    let state = {
    Gujarat : Gujarat
    Rajasthan : Rajasthan
    };
    console.log(state, typeof state);
    ans : error

    eg :

    let state = [
    Gujarat : 1
    Rajasthan : 2
    ];
    console.log(state, typeof state);
    ans : error

  2. Array

    3. eg :

    let sub = ["english", "gujarati", "hindi"];
    console.log(sub, typeof sub);
    ans : (3){'english','gujarati','hindi'} 'object'

    eg :

    let sub = ["english", "gujarati", "hindi"];
    console.log(Array.isArray(sub));
    ans : true

    eg :

    let sub = [english, gujarati, hindi];
    console.log(sub, typeof sub);
    ans : error
